CREATE TABLE [dbo].[IP] (
    [Id]       INT          IDENTITY (1, 1) NOT NULL,
    [IP]       VARCHAR (50) NOT NULL,
    [IdMembre] INT          NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);


CREATE TABLE [dbo].[Visites] (
    [Id]      INT  IDENTITY (1, 1) NOT NULL,
    [IdIP]    INT  NOT NULL,
    [Date]    DATE NOT NULL,
    [NbreVue] INT  NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);
